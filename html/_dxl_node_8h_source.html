122  inline void parseRX(uint8_t c) {
123  // packet = 0xFF,0xFF,ID,N+2,INST_ERR,<N BYTES>,CHECKSUM
124  switch (parserState) {
125 
126  case DXL_SEARCH_FIRST_FF:
127  if (c == 0xFF)
128  parserState = DXL_SEARCH_SECOND_FF;
129  break;
130 
131  case DXL_SEARCH_SECOND_FF:
132  if (c == 0xFF) {
133  parserState = DXL_SEARCH_ID;
134  rxChecksum = 0;
135  } else
136  parserState = DXL_SEARCH_FIRST_FF;
137  break;
138 
139  case DXL_SEARCH_ID:
140  rxChecksum += (pktId = c);
141  parserState = DXL_SEARCH_LENGTH;
142  break;
143 
144  case DXL_SEARCH_LENGTH:
145  if ((isMaster || (pktId == myAddress)) && c < DXL_RX_SIZE) {
146  nParams = c-2;
147  rxChecksum += c;
148  paramsSeen = 0;
149  parserState = DXL_SEARCH_COMMAND;
150  } else
151  parserState = DXL_SEARCH_FIRST_FF;
152  break;
153 
154  case DXL_SEARCH_COMMAND:
155  rxChecksum += (pktCmd = c);
156  parserState = DXL_GET_PARAMETERS;
157  break;
158 
159  case DXL_GET_PARAMETERS:
160  if (paramsSeen >= nParams) {
161  if (c == (uint8_t)~rxChecksum) {
162  // callback
163  if (isMaster)
164  responseReceived = true;
165  if (packetReceived != NULL) {
166  packetReceived(pktId, pktCmd, packet);
167  }
168  parserState = DXL_SEARCH_FIRST_FF;
169  } else {
170  parserState = (c == 0xFF) ? DXL_SEARCH_SECOND_FF : DXL_SEARCH_FIRST_FF;
171  }
172  } else {
173  rxChecksum += (packet[paramsSeen++] = c);
174  }
175  break;
176 
177  default:
178  break;
179  }
180  }
